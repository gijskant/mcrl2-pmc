<SpecChi> 		::= ProcSpec(<DecSpec>)

<DecSpec>		::= DecSpec(<VarSpec>+)
                  | Nil

<VarSpec>		::= VarSpec(<VarExpID>+)

<VarExpID>		::= DataVarExprID(<VarID>, <Exp> ) 
				  |	<VarID> 

<VarID>			::= DataVarID(<String>+, <TypeExp>)

<TypeExp>		::= <TypeID>

<TypeID>		::= Type(<String>)

<Expr>			::= Expression(<String>, <TypeID> )
				  | UnaryExpression(<String>, <Expr>, <TypeID> )
				  | BinaryExpression(<String>, <Expr>, <Expr> , <TypeID> )

<StatementSpec>	::= StatementSpec(<Statement>+)

<Statement>		::= Skip
				  | Assignment(<Expr>+,<Expr>+)
